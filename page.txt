function typewriteAll(){
  clearTimeout(typeTimer);
  codeArea.innerHTML='';
  buildLineNums(LINES.length);

  let lineIdx=0;
  let charIdx=0;
  const rows=[];

  /* crear filas vacías */
  LINES.forEach((l,i)=>{
    const row=document.createElement('div');
    row.className='cl'+(l.active?' aline':'');
    codeArea.appendChild(row);
    rows.push(row);
  });

  const SPEED=28; /* ms por carácter */

  function tick(){
    if(lineIdx>=LINES.length) return;
    const line=LINES[lineIdx];
    const rawLen=line.raw.length;

    if(rawLen===0){ lineIdx++;charIdx=0;typeTimer=setTimeout(tick,SPEED*3);return; }

    charIdx++;

    if(charIdx<=rawLen){
      /* mostrar HTML coloreado hasta charIdx caracteres visibles */
      rows[lineIdx].innerHTML = sliceColoredHTML(line.html, charIdx);
      /* añadir cursor parpadeante al final */
      rows[lineIdx].innerHTML += '<span class="cursor"></span>';
      typeTimer=setTimeout(tick, SPEED);
    } else {
      /* línea completa — mostrar sin cursor */
      rows[lineIdx].innerHTML = line.html;
      lineIdx++;charIdx=0;
      /* pequeña pausa antes de la siguiente línea */
      typeTimer=setTimeout(tick, SPEED*8);
    }
  }
  typeTimer=setTimeout(tick, 400);
}

/* Devuelve el HTML coloreado pero sólo mostrando los primeros `n` chars visibles */
function sliceColoredHTML(html, n){
  /* Parsear los spans y contar sólo texto real */
  const div=document.createElement('div');div.innerHTML=html;
  let visible=0; let result='';

  function processNode(node){
    if(visible>=n) return;
    if(node.nodeType===3){ /* texto */
      const remaining=n-visible;
      const slice=node.textContent.slice(0,remaining);
      result+=escHtml(slice);
      visible+=slice.length;
    } else if(node.nodeType===1){
      const tag=node.tagName.toLowerCase();
      const cls=node.getAttribute('class')||'';
      result+=`<${tag}${cls?' class="'+cls+'"':''}>`;
      node.childNodes.forEach(processNode);
      result+=`</${tag}>`;
    }
  }

  /* también manejar &nbsp; que no son nodos de texto normales */
  /* los &nbsp; en el html origen son literales — los contamos como 1 char cada uno */
  /* usamos innerHTML para parsear correctamente */
  div.childNodes.forEach(processNode);
  return result;
}

function escHtml(s){
  return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function renderTerm(){
  const body = document.getElementById('termBody');
  if(!body) return;
  body.innerHTML = '';
  let timers = [];
  TERM_LINES.forEach((line, idx)=>{
    const div = document.createElement('div');
    div.className = 'tline';
    body.appendChild(div);
    const t = setTimeout(()=>{
      div.innerHTML = '';
      line.parts.forEach(p=>{
        const s = document.createElement('span');
        s.className = p.c;
        s.textContent = p.t;
        div.appendChild(s);
      });
      /* cursor en la última línea */
      /* cursor siempre en la línea activa actual */
      const cur = document.createElement('span');
      cur.className = 'term-cursor';
      cur.id = 'termCursor';
      div.appendChild(cur);
      /* quitar cursor de línea anterior */
      const prev = document.getElementById('termCursor');
      if(prev && prev !== cur) prev.remove();
    }, line.delay);
    timers.push(t);
  });
  /* reiniciar en loop */
  const loopT = setTimeout(()=>{
    timers.forEach(clearTimeout);
    renderTerm();
  }, 9000);
  timers.push(loopT);
}

function openPopup(key, cardEl){

/* ── reveal ── */
(()=>{
  const obs=new IntersectionObserver(es=>{
    es.forEach(e=>{if(e.isIntersecting){e.target.classList.add('vis');obs.unobserve(e.target);}});
  },{threshold:0.08});
  ['pj-label','pj-title','pjWrap'].forEach(id=>{const el=document.getElementById(id);if(el)obs.observe(el);});
})();

/* ── carousel ── */
(()=>{
  const vp    = document.getElementById('pjViewport');
  const track = document.getElementById('pjTrack');
  const dotsEl= document.getElementById('pjDots');
  const prevB = document.getElementById('pjPrev');
  const nextB = document.getElementById('pjNext');
  const currEl= document.getElementById('pjCurr');
  const totEl = document.getElementById('pjTotal');
  if(!track) return;

  const cards = Array.from(track.querySelectorAll('.pj'));
  const N     = cards.length;
  totEl.textContent = N;
  let idx=0, dragging=false, startX=0, startOff=0, liveOff=0;

  /* gap from CSS: 1.4rem. Read actual computed gap */
  function gap(){ return parseFloat(getComputedStyle(track).gap)||22; }
  function cardW(){ return cards[0].getBoundingClientRect().width + gap(); }
  function maxOff(){ return (N-1)*cardW(); }

  /* dots */
  cards.forEach((_,i)=>{
    const d=document.createElement('div');
    d.className='pj-dot'+(i===0?' on':'');
    d.onclick=()=>go(i);
    dotsEl.appendChild(d);
  });

  function go(i, instant=false){
    idx=Math.max(0,Math.min(i,N-1));
    liveOff=idx*cardW();
    if(instant) track.classList.add('snap-off');
    track.style.transform=`translateX(${-liveOff}px)`;
    if(instant) requestAnimationFrame(()=>requestAnimationFrame(()=>track.classList.remove('snap-off')));
    dotsEl.querySelectorAll('.pj-dot').forEach((d,j)=>d.classList.toggle('on',j===idx));
    currEl.textContent=idx+1;
    prevB.disabled=idx===0;
    nextB.disabled=idx===N-1;
  }

  prevB.onclick=()=>go(idx-1);
  nextB.onclick=()=>go(idx+1);

  /* drag helpers */
  function dStart(x){
    dragging=true; startX=x; startOff=liveOff;
    vp.classList.add('is-dragging');
    track.classList.add('snap-off');
  }
  function dMove(x){
    if(!dragging) return;
    const raw=startOff-(x-startX);
    /* rubber-band at edges */
    const clamped=Math.max(-40, Math.min(raw, maxOff()+40));
    liveOff=clamped;
    track.style.transform=`translateX(${-liveOff}px)`;
  }
  function dEnd(x){
    if(!dragging) return;
    dragging=false;
    vp.classList.remove('is-dragging');
    track.classList.remove('snap-off');
    const dx=x-startX;
    const thr=cardW()*0.2;
    if(dx<-thr) go(idx+1);
    else if(dx>thr) go(idx-1);
    else go(idx);
  }

  /* mouse */
  vp.addEventListener('mousedown', e=>{e.preventDefault();dStart(e.clientX);});
  window.addEventListener('mousemove', e=>{dMove(e.clientX);});
  window.addEventListener('mouseup',   e=>{dEnd(e.clientX);});

  /* touch */
  vp.addEventListener('touchstart', e=>dStart(e.touches[0].clientX),{passive:true});
  vp.addEventListener('touchmove',  e=>dMove(e.touches[0].clientX), {passive:true});
  vp.addEventListener('touchend',   e=>dEnd(e.changedTouches[0].clientX),{passive:true});

  /* keyboard */
  document.addEventListener('keydown',e=>{
    if(e.key==='ArrowRight') go(idx+1);
    if(e.key==='ArrowLeft')  go(idx-1);
  });

  /* recalc on resize */
  let resizeT;
  window.addEventListener('resize',()=>{
    clearTimeout(resizeT);
    resizeT=setTimeout(()=>go(idx,true),120);
  });

  go(0);
})();
